var routes = [
     {path:'',component:{template:'<div>这是默认的吃的....</div>'}},
     {
         path:'/food',
         component:food,
         children:[ //子路由不能写/,/代表的是根路径，匹配规则为 /food/meat

             {path:'greens',component:{template:'<div>蔬菜</div>'}},
         ]
     },
     {
         path:'/drink',
         component:drink,
         children:[ // /drink/type/xxx
             {path:'type/:id',component:type}
         ]
     },
     {path:'*',redirect:'/food'}
     //路由是从上到下匹配的，到最下面都没匹配到会走到* 跳转到/food上
];
//默认页面首页router-view是空的 1）配置一个默认页面  2）直接跳转到食物路由上
var food = {template:'#food'};
var drink = {template:'#drink'};
var type = {
    //路径参数 在路径中有随机值  /drink/1/2  /drink/:name/:age  params:{name:1,age:2}
    //获取路由的信息，因为路由已经被注入到vm中
    // 访问的路径/drink/type/3  匹配的规则/drink/type/:id => params:{id:3}
    // this.$route(这里存放的都是关于路由的信息)  this.$route.params.id
    // this.$router(这里存放的是路由的方法)
    template:'<div>当前中第{{id}}</div>',
    computed:{
        id(){
            return this.$route.params.id
        }
    }
};

vue动画
思想:
          vue会transition这个标签会监控里面标签变化的状态,只要状态改变就会触发自己实现的功能
vue单独就可以实现动画效果
步骤
1安装并引入引入vue
  npm install vue --save 
  <script src="node_modules/vue/dist/vue.js"></script>
 2创建vue实例
 <script>
    var vm = new Vue({
        data:{
           flag:false
        },
        el:'#app'
    });
</script>
3 写vue动画的环境
div id="app">
    <button @click="flag=!flag">{{flag?'隐藏':'显示'}}</button>
    <transition>
     <!--transition只支持单动画过渡-->
        <div v-show="flag" class="box"></div>
    </transition>
    <!--name可以改变样式类型的前缀-->
    <transition name="scale">
        <!--transition只支持单动画过渡-->
        <div v-show="flag" class="box1"></div>
    </transition>
</div>
4写vue的css动画属性
<style>
    .box{width: 200px;height: 200px;background: red}
    .box1{width: 200px;height: 200px;background: burlywood}
    .v-enter-active{-webkit-transition: all 1s; opacity: 1;}/*过渡态*/
    .v-enter{opacity: 0;} /*初始态*/
    .v-leave{opacity: 1;} /*过渡状态*/
    .v-leave-active{opacity: 0;-webkit-transition: all 1s;}/*切换状态状态*/
    .scale-enter-active{-webkit-transition: all 2s; opacity: 1;-webkit-transform: scale(1)}
    .scale-enter{opacity: 0;-webkit-transform: scale(0)}
    .scale-leave-active{opacity: 0;-webkit-transition: all 2s;-webkit-transform: scale(0)}
</style>

3创建vm对象
var vm = new Vue({
    //想控制导航，只能在vm中控制
    data:{
       login:true //根据这个属性来控制导航??
    },
    methods:{
      checkLogin(){ //如果直接访问profile没法阻止，应该在进入路由的时候，判断一下如果没登录，不进行跳转，跳转到登录页
          this.login = localStorage.getItem('login') || false;
      },
      logout(){ //点击退出触发这个事件 
          this.login = false;  //将这属性置为false
          this.$router.push('/home'); //跳转到首页
          localStorage.clear('login'); //清除本地存储
      }
    },
    mounted(){
       this.checkLogin();//页面一加载就触发这个方法,初始化,如果已经登录过,置为true,如果没有登录置为false
    },
    el:'#app', //挂载
    router //映射路由
});

//轮播图组件
var slider = {
    data(){
      //自己添加的数据 index轮播图位置 timer定时器的返回值 show:是否显示
      return {index:0,timer:null,show:true}
    },
    computed:{ 计算属性
      prev(){
          //prev是根据 index来的 永远是index的上一个，如果已经是0，应该是最后一个
          if(this.index == 0){ //如果已经是第一个了,那么让这个index到最后一个位置
              return this.datas.length - 1;
          }else{
              return this.index -1;//这个值用永远都是index的上一个 例如index为1 name这个值就是 0
          }
      },
      next(){ //下一个值 比如说index 0  next 1
          if(this.index == this.datas.length - 1){//如果已经是最后一张图,那么显示第一张
              return 0;
          }else{
              return this.index + 1;//用永远都是index的下一个值 
          }
      }
    },
    mounted(){//页面加载完成后开始自动轮播
       this.autoplay();
    },
    methods:{
      go(index){ //点击或者轮播图触发这个方法,获取到当前索引将当前索引保存起开
          this.show = true;
          setTimeout(()=>{
              this.index = index;
              this.show = false
          },100);  
      },
      clear(){
        clearInterval(this.timer); //鼠标离开时触发这个方法,清除定时器
      },
      autoplay(){
          this.timer = setInterval(()=>{
              this.go(this.next);//去下一张
          },this.delay);
      }
    },
    template:'#slider',
    props:{
        datas:{type:Array,required:true},  //获取父亲(home主页)传递给的数据 :datas:""
        delay:{type:Number,default:5000}   //定时器多久执行一次,默认时间5s
    }
};

        <!--show的默认值应该是false-->
        <div class="slider-inner">
            <transition name="out">
                <!--当前显示true index 0  show变为false  离开动画-->
                <img :src="datas[index].path" :alt="datas[index].title" v-if="show">
            </transition>
            <transition name="in">
                <!--index变为1 false->show 进入动画-->
                <img :src="datas[index].path" :alt="datas[index].title" v-if="!show">
            </transition>
        </div>
        <span class="left" @click="go(prev)">&lt;</span>
        <span class="right" @click="go(next)">&gt;</span>
        <div class="slider-bottom">
            <a v-for="(item,index) of datas" @click="go(index)">{{index+1}}</a>
        </div>
    </div>
</template>
